 24 // cache for discovery results

 27 var disc_app_cache_get		= function(app_suffix){ return disc_app_cache[app_suffix];	}

 31 var app_available	= function(app_suffix){
 32 	if( !disc_app_cache_contain(app_suffix) )		return false;
 33 	if( disc_app_cache_get(app_suffix).version == false )	return false;

 37 /**
 38  * Discover an neoip application
 39  * 
 40  * @param {String} app_suffix the neoip application suffix
 41  * @param {function(root_url, version)} success_cb notified if app is found
 42  * @param {function(error)} failure_cb notified if app is not found
 43  * @param {String} hostname the destination hostname to probe
 44 */
 45 var discover_app	= function(ctor_opts){
 46 	// copy ctor_opts + set default values if needed
 47 	var app_suffix	= ctor_opts.app_suffix	|| console.assert(ctor_opts.app_suffix);
 48 	var success_cb	= ctor_opts.success_cb	|| function(){};
 49 	var failure_cb	= ctor_opts.failure_cb	|| function(){};
 50 	var nocache	= ctor_opts.nocache	|| false;
 51 	var verbose	= ctor_opts.verbose	|| 0;
 52 	var hostname	= ctor_opts.hostname	|| "127.0.0.1";
 53 	// sanity check
 54 	console.assert(app_suffix == "oload" || app_suffix == "casti" || app_suffix == "casto");
 55 	// handle cache
 56 	if(app_suffix in disc_app_cache && !nocache ){
 57 		var cache_item	= disc_app_cache[app_suffix];
 58 		if( cache_item.version === false ){
 59 			setTimeout(function(){failure_cb(true);}, 0);
 60 		}else{
 61 			setTimeout(function(){success_cb(cache_item.root_url, cache_item.version)}, 0);			
 62 		}
 63 		return;
 64 	}
 65 	// get info from app_infos
 66 	var port_beg	= app_infos[app_suffix]["port_beg"];
 67 	var port_end	= app_infos[app_suffix]["port_end"];
 68 	var port_cur	= port_beg;
 69 	// define the callbacks
 70 	var probe_succ_cb	= function(version){
 71 		if( verbose )	console.log("found "+app_suffix+" version "+version+" at port "+port_cur);
 72 		var root_url	= "http://"+hostname+":"+port_cur;
 73 		// cache the result
 74 		disc_app_cache[app_suffix]	= {
 75 			"root_url"	: root_url,
 76 			"version"	: version
 77 		};
 78 		// notify the caller
 79 		success_cb(root_url, version);		
 80 	};
 81 	var probe_fail_cb	= function(had_error){
 82 		if( verbose )	console.log(app_suffix+' not found port_cur='+port_cur);
 83 		if(port_cur == port_end){
 84 			// cache the result
 85 			disc_app_cache[app_suffix]	= { "version"	: false	};
 86 			// report "not found" when all port has been tested
 87 			failure_cb("not found");
 88 		}else{
 89 			// test the next port
 90 			port_cur++;
 91 			probe_launch();
 92 		}
 93 	};
 94 	var probe_launch	= function(){
 95 		rpc_call.create({
 96 			call_url	: "http://"+hostname+":"+port_cur+"/neoip_"+app_suffix+"_appdetect_jsrest.js",
 97 			method_name	: 'probe_apps',
 98 			method_args	: [],
 99 			success_cb	: probe_succ_cb,
100 			failure_cb	: probe_fail_cb
101 		});
102 	}
103 	// start the probbing
104 	probe_launch();
105 }

116 /**
117  * Compare version ala memcmp. format is major.minor.patch
118  * - used in discover_webpeer()
119  * - TODO support semver http://semver.org/
120 */
121 var version_compare	= function(version1, version2){
122 	// parse the versions
123 	var matches1	= version1.match(/(\d+).(\d+).(\d+)/);
124 	var matches2	= version2.match(/(\d+).(\d+).(\d+)/);
125 	// compare the major
126 	var major1	= parseInt(matches1[1], 10);
127 	var major2	= parseInt(matches2[1], 10);
128 	if( major1 > major2 )	return +1;
129 	if( major1 < major2 )	return -1;
130 	// compare the minor
131 	var minor1	= parseInt(matches1[2], 10);
132 	var minor2	= parseInt(matches2[2], 10);
133 	if( minor1 > minor2 )	return +1;
134 	if( minor1 < minor2 )	return -1;
135 	// compare the patch
136 	var patch1	= parseInt(matches1[3], 10);
137 	var patch2	= parseInt(matches2[3], 10);
138 	if( patch1 > patch2 )	return +1;
139 	if( patch1 < patch2 )	return -1;
140 	// return 0, they are considered equal
141 	return 0;
142 }

145 // defined the minimal version for each apps
146 var webpeer_versions_min	= {
147 	oload	: "0.0.1",
148 	casto	: "0.0.1",
149 	casti	: "0.0.2"
150 };
151 
152 /**
153  * Possible value of the status
154  * - "toinstall" = not even present
155  * - "toupgrade" = present but not the minimal version
156  * - "installed" = present and uptodate
157  *
158  * @returns {string} return the status of webpeer, or null if not in the cache
159 */
160 var webpeer_status	= function(){
161 	var versions_min	= webpeer_versions_min;
162 	// test if all the apps got probed
163 	for(var app_suffix in versions_min){
164 		if( !(app_suffix in disc_app_cache) )	return null;	
165 	}
166 	// test if all the apps got probed
167 	for(var app_suffix in versions_min){
168 		var version	= disc_app_cache[app_suffix].version;
169 		if( version === false )	return "toinstall";
170 	}
171 	// test if all the apps got probed
172 	for(var app_suffix in versions_min){
173 		var version_cur	= disc_app_cache[app_suffix].version;
174 		var version_min	= versions_min[app_suffix];
175 		if( version_compare(version_cur, version_min) < 0 )	return "toupgrade";
176 	}
177 	// notify the caller
178 	return "installed";
179 }

181 /**
182  * Discover webpeer on localhost and notify the result
183  * 
184  * @param callback {Function} callback notified "toinstall", "toupgrade", "installed"
185 */
186 var discover_webpeer	= function(ctor_opts){
187 	// copy ctor_opts + set default values if needed
188 	var completed_cb= ctor_opts.completed_cb	|| function(){};
189 	var hostname	= ctor_opts.hostname		|| "127.0.0.1";
190 	var nocache	= ctor_opts.nocache		|| false;
191 	// internal vars
192 	var nprobe	= 3;
193 	var callback	= function(){
194 		nprobe	-= 1;
195 		if( nprobe > 0 )	return;
196 		var status	= webpeer_status();
197 		completed_cb(status);
198 	}
199 	// launch the discovery of each app
200 	for(var app_suffix in webpeer_versions_min){
201 		discover_app({
202 			app_suffix	: app_suffix,
203 			hostname	: hostname,
204 			nocache		: nocache,
205 			success_cb	: callback,
206 			failure_cb	: callback
207 		})
208 	}
209 }
