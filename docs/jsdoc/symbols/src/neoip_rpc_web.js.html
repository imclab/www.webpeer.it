  2 // import required modules
  3 var jsonp_call	= require("neoip_jsonp_call").jsonp_call;
  4 
  5 /**
  6  * do a rpc call to a neoip application 
  7 */
  8 var rpc_call = function(ctor_opts){
  9 	//////////////////////////////////////////////////////////////////////////
 10 	//		class variables						//
 11 	//////////////////////////////////////////////////////////////////////////
 12 	// copy ctor_opts + set default values if needed
 13 	var call_url	= ctor_opts.call_url	|| console.assert(ctor_opts.call_url);
 14 	var method_name	= ctor_opts.method_name	|| console.assert(ctor_opts.method_name);
 15 	var method_args	= ctor_opts.method_args	|| [];
 16 	var success_cb	= ctor_opts.success_cb	|| function(returned_val){};
 17 	var failure_cb	= ctor_opts.failure_cb	|| function(error){};
 18 	var verbose	= ctor_opts.verbose	|| 0;
 19 
 20 	//////////////////////////////////////////////////////////////////////////
 21 	//		ctor/dtor						//
 22 	//////////////////////////////////////////////////////////////////////////
 23 	var ctor	= function(){
 24 		client_start();
 25 	}
 26 	var dtor	= function(){
 27 		//console.log("Stop neoip rpc call");
 28 		client_stop();
 29 	}
 30 
 31 	//////////////////////////////////////////////////////////////////////////
 32 	//		http client						//
 33 	//////////////////////////////////////////////////////////////////////////
 34 	var client_call		= null;
 35 	var client_start	= function(){
 36 		// sanity check
 37 		console.assert( client_call == null );
 38 		// build url_path
 39 		var url	= call_url + "?callback=?&method_name=" + method_name;
 40 		for(var i = 0; i < method_args.length; i++){
 41 			url	+= "&arg"+i+"=" + escape(method_args[i]);
 42 		}
 43 		if( verbose > 1 )	console.log("url="+url);
 44 		// create the jsonp_call
 45 		client_call	= new jsonp_call({
 46 			url		: url,
 47 			success_cb	: function(data){
 48 				if( verbose > 1 )	console.dir(data);
 49 				// handle faillure/success_cb at the call level
 50 				if( data['fault'] ){
 51 					failure_cb(data['fault']);	// TODO what about the error itself	
 52 				}else{
 53 					var returned_val= data['returned_val'];
 54 					success_cb(returned_val);
 55 				}				
 56 			},
 57 			failure_cb	: function(error){
 58 				// log to debug
 59 				if( verbose )	console.log("failed due to "+error);
 60 				// notify the caller
 61 				failure_cb(error);
 62 			}
 63 		})
 64 
 65 	}	
 66 	var client_stop		= function(){
 67 		// destroy client_call if needed
 68 		if( client_call )	client_call.destroy();
 69 		client_call	= null;
 70 	}
 71 	
 72 	//////////////////////////////////////////////////////////////////////////
 73 	//		run initialisation					//
 74 	//////////////////////////////////////////////////////////////////////////
 75 	// call the contructor
 76 	ctor();
 77 	// return the public properties
 78 	return {
 79 		destroy	: dtor
 80 	}
 81 };
 82 
 83 /**
 84  * Class method to create an object
 85  * - thus avoid new operator
 86 */
 87 rpc_call.create	= function(ctor_opts){	return new rpc_call(ctor_opts);	}
 88 
 89 // exports public functions
 90 exports.rpc_call	= rpc_call;
