1 #!/usr/bin/env node
  2 
  3 // import the required dependancies
  4 var casti_ctrl_t	= require('./casti_ctrl_t');
  5 
  6 //////////////////////////////////////////////////////////////////////////////////
  7 //	parse cmdline								//
  8 //////////////////////////////////////////////////////////////////////////////////
  9 // cmdline_opts default
 10 cmdline_opts	= {
 11 	casti_opts		: {},
 12 	call_url		: null,
 13 	req_timer_delay		: null,
 14 	verbose			: 0,
 15 	no_gracefull_shutdown	: false
 16 };
 17 var disp_usage	= function(prefix){
 18 	if(prefix)	console.log(prefix + "\n");
 19 	console.log("usage: casti_ctrl -o key val -u url [-d msec] [-n] [-v [-v]]");
 20 	console.log("");
 21 	console.log("Control neoip-casti");
 22 	console.log("");
 23 	console.log("-o|--casti_opts key val\t\tSet the key/val options for casti call.");
 24 	console.log("-u|--call_url url\t\tSet the call_url for the rpc");
 25 	console.log("-d|--req_timer_delay sec\tSet the delay between 2 request_stream calls.");
 26 	console.log("-n|--no_gracefull_shutdown\tDont explicitly release the stream on SIGINT");
 27 	console.log("-v|--verbose\t\t\tIncrease the verbose level (for debug).");
 28 	console.log("-h|--help\t\t\tDisplay the inline help.");
 29 }
 30 var optind	= 2;
 31 for(;optind < process.argv.length; optind++){
 32 	var key	= process.argv[optind];
 33 	var val	= process.argv[optind+1];
 34 	//console.log("key="+key+" val="+val);
 35 	if( key == '-o' || key == "--casti_opts" ){
 36 		cmdline_opts.casti_opts[val]	= process.argv[optind+2];
 37 		optind		+= 2;
 38 	}else if( key == '-u' || key == "--call_url" ){
 39 		// TODO if call_url unspecified, try to autodetect casti on localhost			
 40 		cmdline_opts.call_url	= val;
 41 		optind		+= 1;
 42 	}else if( key == '-d' ||key == "--req_timer_delay" ){
 43 		cmdline_opts.req_timer_delay	= parseFloat(val)*1000;
 44 		optind		+= 1;
 45 	}else if( key == '-n' || key == "--no_gracefull_shutdown" ){
 46 		cmdline_opts.no_gracefull_shutdown	= true;
 47 	}else if( key == '-v' || key == "--verbose" ){
 48 		cmdline_opts.verbose	+= 1;
 49 	}else if( key == "-h" || key == "--help" ){
 50 		disp_usage();
 51 		process.exit(0);
 52 	}else{
 53 		// if the option doesnt exist, consider it is the first non-option parameters
 54 		break;
 55 	}
 56 }
 57 
 58 if(false){
 59 	// example of cmdline
 60 	// Start the original stream
 61 	// $ node node-chargen.js
 62 	// tell webpeer to read this stream:
 63 	// $ node casti_ctrl_t.js --call_url http://localhost:4570/neoip_casti_ctrl_wpage_jsrest.js -o mdata_srv_uri http://localhost/~jerome/neoip_html/cgi-bin/cast_mdata_echo_server.fcgi -o cast_name superstream -o cast_privtext supersecret -o scasti_uri http://127.0.0.1:8124 -o scasti_mod raw -o http_peersrc_uri '' -o web2srv_str 'dummyweb2serv_str' -v
 64 	// Read the stream from webpeer
 65 	// curl `node casto_url_builder.js http://127.0.0.1:4560 a761ce3a superstream`
 66 	cmdline_opts.call_url	= "http://localhost:4570/neoip_casti_ctrl_wpage_jsrest.js";
 67 	cmdline_opts.casti_opts	= {
 68 		mdata_srv_uri	: "http://localhost/~jerome/neoip_html/cgi-bin/cast_mdata_echo_server.fcgi",
 69 		cast_name	: "superstream",
 70 		cast_privtext	: "supersecret",
 71 		scasti_uri	: "http://127.0.0.1:8124",
 72 		scasti_mod	: "raw",
 73 		http_peersrc_uri: "",
 74 		web2srv_str	: "dummyuserdata"
 75 	};
 76 }
 77 
 78 // init casti_ctrl
 79 var casti_ctrl	= casti_ctrl_t.create({
 80 	call_url	: cmdline_opts.call_url,
 81 	casti_opts	: cmdline_opts.casti_opts,
 82 	event_cb	: function(event_type, event_data){
 83 		console.log("event_cb: type="+event_type+" data="+require('sys').inspect(event_data));
 84 		console.log("casti_ctrl: published()="+casti_ctrl.published()+" cast_privhash()="+casti_ctrl.cast_privhash());
 85 	},
 86 	req_timer_delay	: cmdline_opts.req_timer_delay	|| null,
 87 	verbose		: cmdline_opts.verbose		|| 0
 88 });
 89 
 90 // init gracefull_shutdown if not specified otherwise in cmdline
 91 if( cmdline_opts.no_gracefull_shutdown === false ){
 92 	// trap SIGINT - first = release(), second normal behavior
 93 	process.on('SIGINT', function(){
 94 		console.log("Received sigint, start releasing the stream");
 95 		casti_ctrl.release();
 96 		process.removeListener('SIGINT', arguments.callee);
 97 	});
 98 }
