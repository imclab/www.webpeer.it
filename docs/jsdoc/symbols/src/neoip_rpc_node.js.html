1 var http	= require('http');
  2 
  3 /**
  4  * do a rpc call to a neoip application 
  5 */
  6 var rpc_call = function(ctor_opts){
  7 	//////////////////////////////////////////////////////////////////////////
  8 	//		class variables						//
  9 	//////////////////////////////////////////////////////////////////////////
 10 	// alias 'this' for this object, to self
 11 	var self	= this;
 12 	// copy ctor_opts + set default values if needed
 13 	var call_url	= ctor_opts.call_url	|| console.assert(ctor_opts.call_url);
 14 	var method_name	= ctor_opts.method_name	|| console.assert(ctor_opts.method_name);
 15 	var method_args	= ctor_opts.method_args	|| [];
 16 	var success_cb	= ctor_opts.success_cb	|| function(){};
 17 	var failure_cb	= ctor_opts.failure_cb	|| function(){};
 18 	var verbose	= ctor_opts.verbose	|| 0;
 19 
 20 	//////////////////////////////////////////////////////////////////////////
 21 	//		ctor/dtor						//
 22 	//////////////////////////////////////////////////////////////////////////
 23 	var ctor	= function(){
 24 		client_start();
 25 	}
 26 	var dtor	= function(){
 27 		//console.log("Stop neoip rpc call");
 28 		client_stop();
 29 	}
 30 
 31 	//////////////////////////////////////////////////////////////////////////
 32 	//		http client						//
 33 	//////////////////////////////////////////////////////////////////////////
 34 	var client_req		= null;
 35 	var client_start	= function(){
 36 		var url		= require('url').parse(call_url);
 37 		// build url_path
 38 		var url_path	= url.pathname + "?method_name=" + method_name;
 39 		for(var i = 0; i < method_args.length; i++){
 40 			url_path	+= "&arg"+i+"=" + escape(method_args[i]);
 41 		}
 42 		// create the request	  
 43 		var client	= http.createClient((url.port||80), url.hostname);
 44 		// bind error cases at the socket level
 45 		client.on("error"	, function(e){ failure_cb({code: -1, string: e.message});});
 46 		client.on("timeout"	, function(e){ failure_cb({code: -1, string: e.message});});
 47 		// create the request
 48 		client_req	= client.request('GET', url_path, {'host': url.host});
 49 		client_req.on('response', function(client_res){
 50 			// log to debug
 51 			if( verbose )	console.log('STATUS: ' + client_res.statusCode);
 52 			if( verbose )	console.log('HEADERS: ' + JSON.stringify(client_res.headers));
 53 			// Handle faillure at http level
 54 			if(client_res.statusCode != 200)
 55 				return failure_cb(new Error("http statuscode="+client_res.statuscode));
 56 			client_res.setEncoding('utf8');
 57 			client_res.on('data', function( reply_json ){
 58 				if( verbose )	console.log('BODY: ' + reply_json);
 59 				// convert reply_json to native data
 60 				var reply_data	= JSON.parse(reply_json);
 61 				// handle faillure/success_cb at the call level
 62 				if( reply_data['fault'] ){
 63 					if( verbose > 1 )	console.dir(reply_data);
 64 					failure_cb(reply_data['fault']);	// TODO what about the error itself	
 65 				}else{
 66 					var returned_val= reply_data['returned_val'];
 67 					success_cb(returned_val);
 68 				}
 69 			});
 70 			return undefined;
 71 		});
 72 		client_req.end();
 73 	}
 74 	var client_stop		= function(){
 75 		client_req.connection.destroy();
 76 	}
 77 	
 78 	//////////////////////////////////////////////////////////////////////////
 79 	//		run initialisation					//
 80 	//////////////////////////////////////////////////////////////////////////
 81 	// call the contructor
 82 	ctor();
 83 	// return the public properties
 84 	return {
 85 		destroy	: dtor
 86 	}
 87 };
 88 
 89 /**
 90  * Class method to create an object
 91  * - thus avoid new operator
 92 */
 93 rpc_call.create	= function(ctor_opts){	return new rpc_call(ctor_opts);	}
 94 
 95 // exports public functions
 96 exports.rpc_call	= rpc_call;
