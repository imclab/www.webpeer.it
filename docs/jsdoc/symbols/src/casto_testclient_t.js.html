1 // import the required dependancies
  2 var http	= require('http');
  3 
  4 /**
  5  * it read the stream from stream_url.
  6  * - it isnt specific to webpeer stream at all
  7  * - it is mostly a debug tool.
  8  *
  9  * opts:
 10  * - opts.stream_url	: url for the stream
 11  * - opts.notify_unit	: the unit to notify recved_data (default to 1024)
 12  * - opts.event_cb	: callback event_cb(event_type, event_data):
 13  *   - "cnx_begin"/null: server connected
 14  *   - "cnx_end"/null: server disconnected
 15  *   - "recved_size"/nunits: when data is received (nunit is the amount of data
 16  *     in 'unit'). it is notified when there is at least one unit to notified.
 17  *   - "idle_timeout"/null: when no data has been received for 
 18 */
 19 var casto_testclient_t	= function(ctor_opts){
 20 	//////////////////////////////////////////////////////////////////////////
 21 	//		class variables						//
 22 	//////////////////////////////////////////////////////////////////////////
 23 	// alias 'this' for this object, to self
 24 	var self	= this;
 25 	// sanity check - all mandatory fields must be present
 26 	console.assert(ctor_opts.stream_url);
 27 	// copy ctor_opts + set default values if needed
 28 	var stream_url		= ctor_opts.stream_url;
 29 	var event_cb		= ctor_opts.event_cb		|| function(event_type, event_data){};
 30 	var notify_unit		= ctor_opts.notify_unit		|| 1024;
 31 	var verbose		= ctor_opts.verbose		|| 0;
 32 	var idle_timer_delay	= ctor_opts.idle_timer_delay	|| 20*1000;
 33 	var max_recved_len	= ctor_opts.max_recved_len	|| null;
 34 	
 35 	//////////////////////////////////////////////////////////////////////////
 36 	//		ctor/dtor						//
 37 	//////////////////////////////////////////////////////////////////////////
 38 	var ctor	= function(){
 39 		client_start();
 40 	}
 41 	var dtor	= function(){
 42 		idle_timer_stop();
 43 		client_stop();
 44 	}
 45 	
 46 	//////////////////////////////////////////////////////////////////////////
 47 	//		idle_timer						//
 48 	//////////////////////////////////////////////////////////////////////////
 49 	var idle_timer_id	= null;
 50 	var idle_timer_start	= function(){
 51 		if( verbose > 1 )	console.log("launch idle_timer timeout in "+idle_timer_delay+"-msec");
 52 		idle_timer_id	= setTimeout(idle_timer_cb, idle_timer_delay);
 53 	}
 54 	var idle_timer_stop	= function(){
 55 		if( idle_timer_id !== null )	clearTimeout(idle_timer_id);
 56 		idle_timer_id	= null;		
 57 	}
 58 	var idle_timer_refresh	= function(){
 59 		if( verbose )	console.log("idle_time_refresh");
 60 		idle_timer_stop();
 61 		idle_timer_start();
 62 	}
 63 	var idle_timer_cb	= function(){
 64 		if( verbose )	console.log("idle timer expired. next in "+idle_timer_delay+"-msec");
 65 		event_cb("error", "idle timeout after "+idle_timer_delay+"-msec");
 66 	}
 67 
 68 	//////////////////////////////////////////////////////////////////////////
 69 	//		http client						//
 70 	//////////////////////////////////////////////////////////////////////////
 71 	var client_req		= null;
 72 	var client_start	= function(){
 73 		var recved_len	= 0;
 74 		var notified_len= 0;
 75 		// create the http client
 76 		var url		= require('url').parse(stream_url);
 77 		var client	= http.createClient((url.port||80), url.hostname);
 78 		// bind error cases at the socket level
 79 		client.on("error"	, function(e){ event_cb("error", e.message); });
 80 		client.on("timeout"	, function(e){ event_cb("error", e.message); });
 81 		// create the request
 82 		client_req	= client.request('GET', url.pathname, {'host': url.host});
 83 		client_req.on('response', function(client_res){
 84 			// log to debug
 85 			if( verbose )	console.log("Connected to "+stream_url);
 86 			// start the idle_timer
 87 			idle_timer_start();
 88 			// Handle faillure at http level
 89 			if(client_res.statusCode != 200){
 90 				event_cb("error", "statusCode="+client_res.statusCode);
 91 				return
 92 			}
 93 			// notify the caller
 94 			event_cb("cnx_begin", null);
 95 			//client_res.setEncoding('utf8');
 96 			client_res.on('data', function( chunk ){
 97 				// refresh idle_timer
 98 				idle_timer_refresh();
 99 				//console.log("chunk len="+chunk.length);
100 				// update recved_len
101 				recved_len	+= chunk.length;
102 				// notify the recved_size in notify_unit
103 				var notified_chunk	= Math.floor(notified_len	/ notify_unit);
104 				var tonotify_chunk	= Math.floor(recved_len		/ notify_unit);
105 				var nb_chunks		= tonotify_chunk - notified_chunk;
106 				if(nb_chunks > 0)	event_cb("recved_size", nb_chunks);
107 				// update notified_len
108 				notified_len		= recved_len;
109 				// notify the caller
110 				if( max_recved_len && recved_len >= max_recved_len )	event_cb("recved_len_maxed", null);
111 			});
112 			client_res.on('end', function(){
113 				// log the event
114 				if( verbose )	console.log("Connection ended");
115 				// notify the caller
116 				event_cb("cnx_closed", null);
117 			}); 
118 		});
119 		client_req.end();		
120 	}
121 	var client_stop		= function(){
122 		client_req.connection.destroy();
123 	}
124 
125 	//////////////////////////////////////////////////////////////////////////
126 	//		run initialisation					//
127 	//////////////////////////////////////////////////////////////////////////
128 	// call the contructor
129 	ctor();
130 	// return the public properties
131 	return {
132 		destroy	: dtor
133 	}
134 }
135 
136 /**
137  * Class method to create an object
138  * - thus avoid new operator
139 */
140 casto_testclient_t.create	= function(ctor_opts){
141 	return new casto_testclient_t(ctor_opts);
142 }
143 
144 // export it via commonjs
145 exports.create	= casto_testclient_t.create;
