1 #!/usr/bin/env node
  2 
  3 // import required dependancies
  4 var app_detect	= require('./neoip_app_detect');
  5 
  6 //////////////////////////////////////////////////////////////////////////////////
  7 //	parse cmdline								//
  8 //////////////////////////////////////////////////////////////////////////////////
  9 // cmdline_opts default
 10 cmdline_opts	= {
 11 	loop_4ever	: false,
 12 	loop_delay	: 1*1000,
 13 	verbose		: 0
 14 };
 15 var disp_usage	= function(prefix){
 16 	if(prefix)	console.log(prefix + "\n");
 17 	console.log("usage: neoip-detect [options] appname [output_type]");
 18 	console.log("");
 19 	console.log("Discover neoip-apps.");
 20 	console.log("Valid appnames are in 'oload', 'casti', 'casto' and 'webpack'.");
 21 	console.log("appname is optionnal and defaults to 'webpack'.");
 22 	console.log("Valid output_type are : 'root_url', 'version' and 'presence'.");
 23 	console.log("output_type is optionnal and defaults to 'presence'.");
 24 	console.log("if appname is 'webpack', 'presence' is the only valid output_type.");
 25 	console.log("");
 26 	console.log("-c|--continue\tContinue for ever, instead of only 1 iteration.");
 27 	console.log("-d|--delay num\tSet the loop delay (valid only with --continue).");
 28 	console.log("-v|--verbose\tIncrease the verbose level (for debug).");
 29 	console.log("-h|--help\tDisplay the inline help.");
 30 }
 31 var optind	= 2;
 32 for(;optind < process.argv.length; optind++){
 33 	var key	= process.argv[optind];
 34 	var val	= process.argv[optind+1];
 35 	//console.log("key="+key+" val="+val);
 36 	if( key == '-c' || key == "--continue" ){
 37 		cmdline_opts.loop_4ever	= true;
 38 	}else if( key == '-d' || key == "--delay" ){
 39 		cmdline_opts.loop_delay	= parseFloat(val)*1000;
 40 		optind++;
 41 	}else if( key == '-v' || key == "--verbose" ){
 42 		cmdline_opts.verbose	+= 1;
 43 	}else if( key == "-h" || key == "--help" ){
 44 		disp_usage();
 45 		process.exit(0);
 46 	}else{
 47 		// if the option doesnt exist, consider it is the first non-option parameters
 48 		break;
 49 	}
 50 }
 51 	
 52 // get required options from the rest of the cmdline
 53 var app_name	= process.argv[optind++] || 'webpack';
 54 var output_type	= process.argv[optind++] || 'presence';
 55 
 56 var loop_4ever	= cmdline_opts.loop_4ever;
 57 var loop_delay	= cmdline_opts.loop_delay;
 58 var verbose	= cmdline_opts.verbose;
 59 
 60 // sanity check - app_name MUST be in valid_names
 61 var valid_names = ['oload', 'casti', 'casto', 'webpack'];
 62 if( valid_names.indexOf(app_name) == -1 ){
 63 	console.log("app_name '"+app_name+"' is not possible.");
 64 	console.log("Valid names are ", valid_names);
 65 	process.exit(-1);
 66 }
 67 
 68 // sanity check - app_name MUST be in valid_names
 69 var valid_types = ['root_url', 'version', 'presence'];
 70 if( valid_types.indexOf(output_type) == -1 ){
 71 	console.log("output_type '"+output_type+"' is not possible.");
 72 	console.log("Valid types are "+require('sys').inspect(valid_names));
 73 	process.exit(-1);
 74 }
 75 if( app_name == 'webpack' && output_type != 'presence' ){
 76 	console.log("with 'webpack' apps name, 'presence' is the only valid output_type");
 77 	process.exit(-1);
 78 }
 79 
 80 // trace for the user
 81 if( verbose )	console.log("discovering '"+app_name+"' for "+output_type);
 82 
 83 //////////////////////////////////////////////////////////////////////////
 84 //	do actual discovery						//
 85 //////////////////////////////////////////////////////////////////////////
 86 var one_iter	= function(completed_cb){
 87 	if( app_name != 'webpack' ){
 88 		// discover neoip-oload and act depending on its presence or not
 89 		app_detect.discover_app(app_name, function(root_url, version){
 90 			//console.log("root_url="+root_url+ " version="+version);
 91 			if( output_type == 'root_url' )		console.log(root_url);
 92 			else if( output_type == 'version' )	console.log(version);
 93 			else if( output_type == 'presence' )	console.log("present ("+version+")");
 94 			completed_cb(0);
 95 		}, function(reason){
 96 			if( verbose )			console.log("Failed due to "+reason);
 97 			if( output_type == 'presence' )	console.log("not present");
 98 			completed_cb(-1);
 99 		})
100 	}else if( app_name == 'webpack' && output_type == 'presence' ){
101 		app_detect.discover_webpack(function(status){
102 			if( verbose )	console.log("webpack status="+status);	
103 			if( status == "installed" )		console.log("webpack is installed and uptodate.");
104 			else if( status == "toupgrade" )	console.log("webpack is installed and needs upgrade.");
105 			else if( status == "toinstall" )	console.log("webpack is not installed.");
106 			else console.assert(false);
107 			// return the proper statuscode
108 			if( status == "toinstall" )	completed_cb(-1);
109 			else				completed_cb(0);
110 		});
111 	}
112 }
113 
114 if( loop_4ever == false ){
115 	one_iter(function(statuscode){
116 		process.exit(statuscode);
117 	})
118 }else{
119 	// trap SIGINT - exit with error
120 	process.on('SIGINT', function(){
121 		process.exit(-1);
122 	});
123 	var	endless_loop	= function(){
124 		one_iter(function(statuscode){
125 			// clear the cache - force to get fresh result
126 			app_detect.cache_clear();
127 			setTimeout(endless_loop, loop_delay);
128 		})
129 	}
130 	endless_loop();		
131 }

140 =head1 NAME
141 
142 neoip-detect - Detect neoip applications
143 
144 =head1 SYNOPSIS
145 
146 B<neoip-detect>  [options] [appname [output_type]]
147 
148 =head1 DESCRIPTION
149 
150 B<neoip-detect> detect neoip-applications presence and related informations, like
151 version or api's root_url.
152 
153 =over
154 
155 =item I<appname>
156 
157 Valid appnames are in 'oload', 'casti', 'casto' and 'webpack'.
158 It is optionnal and defaults to 'presence'.
159 
160 =item I<output_type>
161 
162 Determine the type of output to use. Valid output_type are : 'root_url', 'version' and 'presence'.
163 It is optionnal and defaults to 'presence'.
164 
165 B<Note>: if I<appname> is 'webpack', the only valid type is 'presence'
166 
167 =over
168 
169 =item I<root_url>
170 
171 It will output in stdout the API's root url for this application.
172 
173 =item I<version>
174 
175 It will output in stdout version this application.
176 
177 =item I<presence>
178 
179 It will display a human readable message on stdout. This option is interesting
180 mainly for its statuscode returned by the executable.
181 
182 =back
183 
184 =back
185 
186 =head1 OPTIONS
187 
188 The following options are available:
189 
190 =over
191 
192 =item B<--c|--continue>
193 
194 It will continue looping for ever.
195 By default, it does only one iteration.
196 
197 =item B<--d|--delay sec>
198 
199 It will set the delay between each loop iteration.
200 This option is valid only with B<--continue>.
201 By default, it is 1-sec. Fractional seconds is supported.
202 
203 =item B<-v|--verbose>
204 
205 Increase the verbose level. It may be used multiple times.
206 
207 =back
208 
209 =head1 EXAMPLES
210 
211 =over
212 
213 =item B<version display>
214 
215 $ neoip-detect oload version
216 
217 This will display the following in stdout
218 
219 0.0.1
220 
221 =item B<detecting presence>
222 
223 $ neoip-detect webpack && echo "detected" || echo "not detected"
224 
225 This will display I<detected> in stdout if neoip-webpack is detected
226 and I<not detected> otherwise.
227 
228 =cut
