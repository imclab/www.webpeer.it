1 var collection	= function(){
  2 	// define private variables
  3 	var _map	= {}
  4 	
  5 	var _parse_key	= function(key, non_exist_callback){
  6 		// parse the key into parts
  7 		var key_parts	= key.split("/");
  8 		// goto the proper submap (and create it if needed)
  9 		var submap	= _map;
 10 		var subkey	= key_parts[0]
 11 		for(var i = 0; i < key_parts.length - 1; i++){
 12 			// if this subkey is not present in submap, notify an exception
 13 			if( typeof submap[subkey] == 'undefined' )	non_exist_callback(submap, subkey);
 14 			// goto the next submap
 15 			submap	= submap[subkey];
 16 			subkey	= key_parts[i+1];
 17 		}
 18 		// return the result
 19 		return {
 20 			'submap': submap,
 21 			'subkey': subkey
 22 		};
 23 	};
 24 	
 25 	/**
 26 	 * Set the variable namespace/key to the value val
 27 	 */
 28 	var set	= function(key, val) {
 29 		// parse the key
 30 		var parsed_key	= _parse_key(key, function(submap, subkey){
 31 			// if this subkey is not present in submap, create an empty object
 32 			submap[subkey]	= {}
 33 		});
 34 		// set this value in the last submap
 35 		parsed_key.submap[parsed_key.subkey]	= val;
 36 	}
 37 	/**
 38 	 * Return the value of the variable namespace/key (note: it MUST be defined)
 39 	*/
 40 	var get	= function(key) {
 41 		// sanity check - the key MUST be present
 42 		console.assert( has(key) );
 43 		// parse the key
 44 		var parsed_key	= _parse_key(key, function(submap, subkey){
 45 			// if this subkey is not present in submap, notify an exception
 46 			throw new Error('subkey '+subkey+' (from key '+key+') doesnt exist');
 47 		});
 48 		// get this value in the last submap
 49 		var val	= parsed_key.submap[parsed_key.subkey];
 50 		// return this val
 51 		return val;
 52 	}
 53 	var get_dfl	= function(key, dfl){ return has(key) ? get(key) : dfl; }
 54 
 55 	
 56 	/**
 57 	 * delete this key (note: it MUST be defined)
 58 	*/
 59 	var del	= function(key) {
 60 		// sanity check - the key MUST be present
 61 		console.assert( has(key) );
 62 		// parse the key
 63 		var parsed_key	= _parse_key(key, function(submap, subkey){	
 64 			// if this subkey is not present in submap, notify an exception
 65 			throw new Error('subkey '+subkey+' (from key '+key+') doesnt exist');
 66 		});
 67 		// get this value in the last submap
 68 		var val	= parsed_key.submap[parsed_key.subkey];
 69 		// delete in the last submap
 70 		delete parsed_key.submap[parsed_key.subkey];
 71 	}
 72 	
 73 	/**
 74 	 * Return true if this variable is defined, false otherwize
 75 	*/
 76 	var has	= function(key) {
 77 		// parse the key
 78 		var parsed_key	= null;
 79 		// TODO i could avoid the exception by doing closure on parsed_key ?
 80 		try {
 81 			parsed_key	= _parse_key(key, function(submap, subkey){
 82 				// if this subkey is not present in submap, notify an exception
 83 				throw new Error('subkey '+subkey+' (from key '+key+') doesnt exist');
 84 			});
 85 		}catch(error) {
 86 			// return false now, if the key cant be parsed
 87 			return	false;		
 88 		}
 89 		// if this subkey is not present in submap, notify an exception
 90 		if( typeof parsed_key.submap[parsed_key.subkey] == 'undefined' ) return false;
 91 		// if all previous tests passed, return true
 92 		return true;
 93 	}
 94 	
 95 	return {
 96 		"set"		: set,
 97 		"get"		: get,
 98 		"get_dfl"	: get_dfl,
 99 		"del"		: del,
100 		"has"		: has
101 	}
102 };
