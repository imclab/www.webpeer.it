1 // import required modules
  2 var collection	= require('./collection').collection;
  3 var base64	= require('./base64').base64;
  4 
  5 var url_builder_oload_t	= function(nested_uri_str){
  6 	var _col	= new collection();
  7 	
  8 	var outter_uri	= function(val){ _col.set('outter_uri'	, val); }
  9 	var inner_uri	= function(val){ _col.set('inner_uri'	, val); }
 10 	var outter_var	= function(key, val){ return _col.set('outter_var/'+key, val);	}
 11 	var minner_var	= function(key, val){ return _col.set('minner_var/'+key, val);	}
 12 	var dupuri	= function(val){
 13 				for(var i = 0; ; i++){
 14 					var key	= "outter_var/dupuri/"+i;
 15 					if( _col.has(key) ) continue;
 16 					_col.set(key, val);
 17 					break;
 18 				}
 19 			}
 20 	
 21 	////////////////////////////////////////////////////////////////////////////////
 22 	////////////////////////////////////////////////////////////////////////////////
 23 	//			Checker function
 24 	////////////////////////////////////////////////////////////////////////////////
 25 	////////////////////////////////////////////////////////////////////////////////
 26 	
 27 	/** \brief throw an exception is this object is not considered sane 
 28 	 */
 29 	var _is_sane_internal	= function() {
 30 		if( !_col.has("outter_uri") )	throw new Error("No outter_uri");
 31 		if( !_col.has("inner_uri") )	throw new Error("No inner_uri");
 32 		
 33 		// TODO do all the sanity check here
 34 		// - if subfile_level exist, a subfile_path MUST too
 35 		// - subfile_path MUST always start with '/'
 36 		// - if 'type' check the value is a legal one
 37 		// - if 'mod' check the value is a legal one
 38 		// - for dupuri and http_peersrc_uri, it MUST start by 'http://'
 39 	}
 40 	
 41 	/** \brief If this object is considered sane, return true. false otherwise
 42 	 */
 43 	var is_sane	= function(){
 44 		try {
 45 			// call the version with exception
 46 			_is_sane_internal();
 47 		}catch(error) {
 48 			console.log("url_builder_oload_t not sane due to " + error);
 49 			return	false;		
 50 		}
 51 		// if all previous tests passed, this is considered sane
 52 		return true;
 53 	}
 54 	
 55 
 56 	////////////////////////////////////////////////////////////////////////////////
 57 	////////////////////////////////////////////////////////////////////////////////
 58 	//			to_string() function
 59 	////////////////////////////////////////////////////////////////////////////////
 60 	////////////////////////////////////////////////////////////////////////////////
 61 	
 62 	/** \brief return a string of the nested_uri
 63 	 */
 64 	var to_string	= function(){
 65 		// some functions pointer for configuration
 66 		//var url_encode_safe	= neoip_base64.encode_safe;
 67 		var _url_encode_safe	= base64.encode_safe;
 68 		var _url_decode_safe	= base64.decode_safe;
 69 		//var url_doscramble	= neoip.core.doscramble_uri;
 70 		var _url_doscamble	= function(url){ return url; }
 71 		// define local variables
 72 		var result	= "";
 73 		// sanity check - the object MUST be sane
 74 		console.assert( is_sane() );
 75 // neoip-url inner_uri
 76 // outter_uri = discovered
 77 // inner_uri  = from cmdline
 78 // --outter_var/-o	key=val
 79 // --minner_var/-i	key=val
 80 //------------------
 81 // --mode/-m		alias for outter_var/mod
 82 // --dupuri/-d  	alias for outter_var/dupuri
 83 // --path/-p		alias for outter_var/subfile_path
 84 // --type/-t		alias for outter_var/link_type
 85 
 86 // {{outter_uri}}/{{mod}}/(*{{outter_var_key}}*{{outter_var_val}}/)*
 87 // /{{inner_uri}}(?@)neoip_metavar_{{minner_var_key}}=#{{minner_var_val}}
 88 
 89 		// start building the nested_uri
 90 		result	+= _col.get('outter_uri') + "/";
 91 		
 92 		// put the 'mod' variable first
 93 		if( _col.has('outter_var/mod') )	result += _col.get('outter_var/mod') + "/";
 94 	
 95 		// put all the outter variables
 96 		for(var key in _col.get_dfl('outter_var', {}) ){
 97 			// skip key equal to dupuri/subfile_path, they are handled separatly
 98 			if( key == 'dupuri' )		continue;
 99 			if( key == 'subfile_path' )	continue;
100 			if( key == 'mod' )		continue;
101 			// put the key of the variable
102 			result	+= "*" + key + "*";
103 			// get the value
104 			var val	= _col.get('outter_var/'+key);
105 			// http_peersrc_uri is specific - values are encoded in base64-urlsafe
106 			if( key == "http_peersrc_uri" )	val = url_encode_safe(val)
107 			// put the values according to the keys
108 			result	+= val;
109 			// add the separator
110 			result	+= "/";
111 		}
112 		
113 		// handle outter_var/subfile_path, aka insert the dynamic outter_var subfile_level
114 		if( _col.has('outter_var/subfile_path') ){
115 			var subfile_path	= _col.get('outter_var/subfile_path');
116 			var subfile_level	= subfile_path.split("/").length - 1;	// put the key of the variable
117 			// add the subfile_level as outter_var in result
118 			result	+= "*subfile_level*"+subfile_level+'/';
119 		}
120 		
121 		// put all the dupuri with value in base64-urlsafe encoding
122 		for(var dupuri_idx in _col.get_dfl('outter_var/dupuri', {})){
123 			result	+= "*dupuri*";
124 			result	+= _url_encode_safe(_col.get('outter_var/dupuri/'+dupuri_idx));
125 			result	+= "/";
126 		}
127 	
128 		// put the inner_uri at the end
129 		// - made complex by the need to put the m_subfile_path between the 
130 		//   path and the query part of the inner_uri
131 		var inner_uri	= _col.get('inner_uri');
132 		var has_subfile	= _col.has('outter_var/subfile_path')
133 		var subfile_path= _col.get_dfl("outter_var/subfile_path", null);
134 		var query_pos	= inner_uri.indexOf("?");
135 		if( query_pos != -1 )	result	+= inner_uri.substr(0, query_pos);
136 		else			result	+= inner_uri;
137 		if( subfile_path )	result	+= subfile_path
138 		if( query_pos != -1 )	result	+= inner_uri.substr(query_pos, inner_uri.length);
139 	
140 		// put all the inner variables aka "neoip_metavar_"
141 		for(var key in _col.get_dfl('minner_var', {}) ){
142 			// put the variable separator
143 			result	+= result.indexOf('?') == -1 ? "?" : "&";
144 			// put the key of the variable
145 			result	+= 'neoip_metavar_' + key + "=" + escape(_col.get('minner_var/'+key));
146 		}
147 		
148 		// scramble the result
149 		result	= _url_doscamble(result);
150 		// return the just built nested_uri
151 		return result;
152 	}
153 
154 	////////////////////////////////////////////////////////////////////////////////
155 	////////////////////////////////////////////////////////////////////////////////
156 	//			from_string() function
157 	////////////////////////////////////////////////////////////////////////////////
158 	////////////////////////////////////////////////////////////////////////////////
159 	/**
160 	 * Parse a string and set _col with it
161 	*/
162 	var from_string	= function(from_str){
163 		// some functions pointer for configuration
164 		//var url_decode_safe	= neoip_base64.decode_safe;
165 		var _url_decode_safe	= base64.decode_safe;
166 		// initialisation of nleft_str/nright_str from nested_str
167 		var nested_str	= from_str;
168 		var nleft_str	= nested_str.substr(0, from_str.indexOf('/http:/'));
169 		var nright_str	= nested_str.substr(from_str.indexOf('/http:/')+1);
170 		// Process outter_var: consume all the outter_var in nleft_str (outter_var/mod included)
171 		while( true ){
172 			// extract last_level from nleft_str
173 			var last_level	= nleft_str.substr(nleft_str.lastIndexOf('/')+1);
174 			// if last_level is a normally encoded outter_var
175 			if( last_level.substr(0, 1) == '*' ){
176 				var matches	= last_level.match(/\*(.+)\*(.+)$/);
177 				var key		= matches[1];
178 				var val		= matches[2];
179 				if( key != "dupuri" )	_col.set('outter_var/'+key, val);
180 				else			dupuri(_url_decode_safe(val));
181 			}else if( last_level == "raw" || last_level == "flv" ){
182 				// if last_level is a outter_var/mod
183 				_col.set('outter_var/mod', last_level)
184 			}else {
185 				// if last_level is not recognized, leave the loop
186 				break;
187 			}
188 			// consume in nleft_str
189 			nleft_str	= nleft_str.substr(0, nleft_str.lastIndexOf('/'));
190 		}
191 		// set "outter_uri" - what remains in nleft_str is 'outter_uri'
192 		_col.set("outter_uri", nleft_str);
193 		// declare the 'actual inner uri variables' array
194 		var ainner_vars	= [];
195 		// if the right part contains variables, process them to extract minner_vars
196 		if( nright_str.lastIndexOf('?') != -1 ){
197 			var search_str	= nright_str.substr(nright_str.lastIndexOf('?')+1);
198 			var keyval_arr	= search_str.split("&");
199 			// go thru each variable
200 			for(var i = 0; i < keyval_arr.length; i++ ){
201 				var keyval	= keyval_arr[i].split("=");
202 				var key		= keyval[0];
203 				var val		= keyval[1];
204 				// if this key is not a minner_var, simply copy it in ainner_vars
205 				if( key.indexOf("neoip_metavar_") != 0 ){
206 					ainner_vars.push(keyval_arr[i]);
207 					continue
208 				}
209 				var minner_key	= key.substr("neoip_metavar_".length);
210 				_col.set("minner_var/"+minner_key, val);
211 			}
212 			// consume the query part of the nright_str
213 			nright_str	= nright_str.substr(0, nright_str.lastIndexOf('?'));
214 		}
215 		// if outter_var/subfile_level is present, handle it here
216 		if( _col.has('outter_var/subfile_level') ){
217 			var subfile_level	= _col.get('outter_var/subfile_level');
218 			// find the begining of the subfile_path
219 			var pos 		= null;
220 			for(var i = 0; i < subfile_level; i++){
221 				if( pos )	pos = nright_str.lastIndexOf('/', pos-1);
222 				else		pos = nright_str.lastIndexOf('/');
223 			}
224 			// extract the subfile_path
225 			var subfile_path	= nright_str.substr(pos);
226 			_col.set("outter_var/subfile_path"	, subfile_path);
227 			// delete outter_var/subfile_level
228 			_col.del("outter_var/subfile_level");
229 			// consume the subfile_path
230 			nright_str	= nright_str.substr(0, pos);
231 		}
232 		// generate the inner_uri
233 		var inner_uri	= nright_str;
234 		// append actual inner variables, if there is any
235 		if( ainner_vars.length > 0 )	inner_uri	+= '?' + ainner_vars.join('&');
236 		// set inner_uri
237 		_col.set('inner_uri'	, inner_uri);		
238 	}
239 	
240 	
241 	// if nested_uri_str is defined, use it for building the uri
242 	if( nested_uri_str !== undefined )	from_string(nested_uri_str);
243 	
244 	
245 	// export the public function
246 	var returned_obj	= {
247 		"outter_uri"	: outter_uri,
248 		"inner_uri"	: inner_uri,
249 		"outter_var"	: outter_var,
250 		"minner_var"	: minner_var,
251 		"dupuri"	: dupuri,
252 		"set"		: function(key, val) { _col.set(key, val); return this;},
253 		"get"		: _col.get,
254 		"get_dfl"	: _col.get_dfl,
255 		"del"		: _col.del,
256 		"has"		: _col.has,
257 		"is_sane"	: is_sane,
258 		"to_string"	: to_string,
259 		"from_string"	: from_string
260 	}
261 	return returned_obj;
262 }

269 url_builder_oload_t.create	= function(ctor_opts){	return new url_builder_oload_t(ctor_opts);	}
