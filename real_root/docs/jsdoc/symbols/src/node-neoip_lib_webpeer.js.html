1 // import required modules
  2 var app_detect	= require('./neoip_app_detect');
  3 var url_builder_oload_t= require('./url_builder_oload_t');
  4 
  5 var verbose	= 1;
  6 
  7 /**
  8  * probe webpeer and notify the callback once completed
  9  * 
 10  * @param completed_cb {function} callback notified on completion completed(avail){}
 11 */
 12 webpeer_ready	= function(completed_cb){
 13 	// discover neoip-oload
 14 	app_detect.webpack_probe(function(status){
 15 		if( status == "toinstall" )	completed_cb(false);
 16 		else if( status == "toupgrade")	completed_cb(true);
 17 		else if( status == "installed")	completed_cb(true);
 18 		else console.assert(false);
 19 	})
 20 };
 21 
 22 
 23 /**
 24  * @returns {boolean} true if webpeer is available, false otherwise
 25 */
 26 webpeer_avail	= function(){
 27 	return app_detect.webpack_status() == "installed";
 28 }
 29 
 30 /**
 31  * webpeerify a url to static file
 32  * - this is an minimal helper on top of url_builder_oload_t
 33  * - url_builder_oload_t implementation is complete and much more complex
 34  * 
 35  * @param url {string} original url to webpeerify
 36 */
 37 webpeer_url	= function(url){
 38 	if( !webpeer_avail() )	return url;
 39 	return url_builder_oload_t.create(url)
 40 			.set('outter_uri', app_detect.cache_get('oload').root_url)
 41 			.to_string();
 42 }
 43 
 44 
 45 // exports public functions
 46 exports.ready	= webpeer_ready;
 47 exports.avail	= webpeer_avail;
 48 exports.url	= webpeer_url;
