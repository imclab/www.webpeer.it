 11 =head1 NAME
 12 
 13 neoip-url-stream - build a url for neoip-casto
 14 
 15 =head1 SYNOPSIS
 16 
 17 B<neoip-url-stream> [-s mdata_srv_uri] base_url cast_privhash cast_name
 18 
 19 =head1 DESCRIPTION
 20 
 21 Build an url for neoip-casto. This url will point on a normal http stream.
 22 Thus the client may be any client which understand http, it doesnt have to modified
 23 to use this url.
 24 
 25 =over
 26 
 27 =item I<base_url>
 28 
 29 The base_url is the neoip-casto base url. It is not autodetected.
 30 TODO make it autodetectable (likely with a neoip-appdetect applications
 31 or by simply make a dependancy in the code)
 32 
 33 =item I<cast_privhash>
 34 
 35 This is the hash of the broadcast password (aka casti_privtext). This is used
 36 to ensure the http client got the permission to read the broadcast.
 37 
 38 =item I<cast_name>
 39 
 40 cast_name is the name of the broadcast. This name is unique inside the I<mdata_srv_uri>.
 41 
 42 =back
 43 
 44 =head1 OPTIONS
 45 
 46 The following options are available:
 47 
 48 =over
 49 
 50 =item B<-s|--mdata_srv_uri> I<url>
 51 
 52 Use to set the url for the metadata server. This server got the metadata information
 53 about the broadcast. (OPTIONAL)
 54 
 55 =back
 56 
 57 =head1 EXAMPLES
 58 
 59 =over
 60 
 61 =item B<basic usage>
 62 
 63 $ neoip-url-stream http://localhost:4560 a761ce3a superstream
 64 
 65 This will display the following in stdout
 66 
 67 http://localhost:4560/a761ce3a/superstream
 68 
 69 =item B<specifying mdata_srv_uri>
 70 
 71 $ neoip-url-stream -s http://example.com http://localhost:4560 a761ce3a superstream
 72 
 73 This will display the following in stdout
 74 
 75 http://localhost:4560/a761ce3a/superstream?mdata_srv_uri=http%3A//example.com
 76 
 77 =cut
 78 
 79 
 80 */

 82 /**
 83  * Create an url for neoip-casto
 84  * - opts.base_url	: casto base_url	(REQUIRED)
 85  * - opts.cast_privhash	: cast_privhash		(REQUIRED)
 86  * - opts.cast_name	: cast_name		(REQUIRED)
 87  * - opts.mdata_srv_uri	: url of the nameserver (OPTIONAL)
 88  *
 89  * - NOTE: no detection of the neoip-casto apps is done
 90  *
 91  * @return the url for the stream out of neoip-casto
 92  * @type String
 93 */
 94 var create = function(opts){
 95 	// sanity check - all mandatory fields MUST be present
 96 	console.assert(opts.base_url);
 97 	console.assert(opts.cast_privhash);
 98 	console.assert(opts.cast_name);
 99 	// build the url
100 	var url	= opts.base_url + "/" + opts.cast_privhash + "/" + opts.cast_name;
101 	// add mdata_srv_uri if any
102 	if( opts.mdata_srv_uri )	url	+= "?mdata_srv_uri=" + escape(opts.mdata_srv_uri);
103 	// return the just built url
104 	return url;
105 }
106 
107 // export it via commonjs
108 exports.create	= create;
109 
110 //////////////////////////////////////////////////////////////////////////////////
111 //	main programm								//
112 //////////////////////////////////////////////////////////////////////////////////
113 if( process.argv[1] == __filename ){
114 	opts	= {
115 		base_url	: null,
116 		cast_privhash	: null,
117 		cast_name	: null,
118 		mdata_srv_uri	: null
119 	};
120 
121 	//////////////////////////////////////////////////////////////////////////////////
122 	//	parse cmdline								//
123 	//////////////////////////////////////////////////////////////////////////////////
124 	var disp_usage	= function(prefix){
125 		if(prefix)	console.log(prefix + "\n");
126 		console.log("usage: neoip-url-stream [-s url] base_url cast_privhash cast_name");
127 		console.log("");
128 		console.log("Build an url for neoip-casto");
129 		console.log("");
130 		console.log("-s|--mdata_srv_uri\n\t\tSet url for the mdata_srv.");
131 		console.log("-h|--help\tDisplay the inline help.");
132 	}
133 	var optind	= 2;
134 	for(;optind < process.argv.length; optind++){
135 		var key	= process.argv[optind];
136 		var val	= process.argv[optind+1];
137 		//console.log("key="+key+" val="+val);
138 		if( key == "-s" || key == "--mdata_srv_uri" ){
139 			opts.mdata_srv_uri	= val;
140 			optind		+= 1;			
141 		}else if( key == "-h" || key == "--help" ){
142 			disp_usage();
143 			process.exit(0);
144 		}else{
145 			// if the option doesnt exist, consider it is the first non-option parameters
146 			break;
147 		}
148 	}
149 	// check there are at least 3 parameters remaining in the cmdline
150 	if( process.argv.length - optind < 3 ){
151 		disp_usage("missing parameters to buidl the url");
152 		process.exit(0);		
153 	}
154 	
155 	// get required options from the rest of the cmdline
156 	opts.base_url		= process.argv[optind++];
157 	opts.cast_privhash	= process.argv[optind++];
158 	opts.cast_name		= process.argv[optind++];
159 	
160 	// create the url
161 	var url	= create(opts);
162 	// output the url on stdout
163 	console.log(url);	
164 }
