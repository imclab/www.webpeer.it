1 var neoip_rpc	= require('./neoip_rpc_node');
  2 var sys		= require('sys');
  3 var underscore	= require('../vendor/underscore/underscore')._; underscore.noConflict();
  4 
  5 /**
  6 */
  7 var casti_ctrl_t	= function(ctor_opts){
  8 	//////////////////////////////////////////////////////////////////////////
  9 	//		class variables						//
 10 	//////////////////////////////////////////////////////////////////////////
 11 	// alias 'this' for this object, to self
 12 	var self		= this;
 13 	// copy ctor_opts + set default values if needed
 14 	var call_url		= ctor_opts.call_url		|| console.assert(false);
 15 	var casti_opts		= ctor_opts.casti_opts		|| console.assert(false);
 16 	var event_cb		= ctor_opts.event_cb		|| function(event_type, event_data){}
 17 	var req_timer_delay	= ctor_opts.req_timer_delay	|| 0.5*1000;
 18 	var verbose		= ctor_opts.verbose		|| 0;
 19 	// private methods
 20 	var cast_privhash	= null;
 21 
 22 	//////////////////////////////////////////////////////////////////////////
 23 	//		ctor/dtor						//
 24 	//////////////////////////////////////////////////////////////////////////
 25 	var ctor	= function(){
 26 		rpc_call_request();
 27 	}
 28 	var dtor	= function(){
 29 		// destroy pending rpc_call if needed
 30 		rpc_call_destroy();
 31 		// stop the req_timer if needed
 32 		req_timer_stop();
 33 	}
 34 	
 35 	//////////////////////////////////////////////////////////////////////////
 36 	//		misc							//
 37 	//////////////////////////////////////////////////////////////////////////
 38 	var start_release	= function(){
 39 		// destroy pending rpc_call if needed
 40 		rpc_call_destroy();
 41 		// stop the req_timer if needed
 42 		req_timer_stop();
 43 		// launch_rpc_call_release
 44 		rpc_call_release();
 45 	}
 46 	
 47 	//////////////////////////////////////////////////////////////////////////
 48 	//		req_timer						//
 49 	//////////////////////////////////////////////////////////////////////////
 50 	var req_timer_id	= null;
 51 	var req_timer_start	= function(delay){
 52 		// if not specified, set delay to default value
 53 		if( delay == undefined )	delay	= req_timer_delay;
 54 		console.assert(req_timer_id === null);
 55 		req_timer_id	= setTimeout(req_timer_cb, delay);
 56 	}
 57 	var req_timer_stop	= function(){
 58 		if( req_timer_id !== null )	clearTimeout(req_timer_id);
 59 		req_timer_id	= null;		
 60 	}
 61 	var req_timer_refresh	= function(){
 62 		req_timer_stop();
 63 		req_timer_start();
 64 	}
 65 	var req_timer_cb	= function(){
 66 		if( verbose > 1 )	console.log("req_timer expired. next in "+req_timer_delay+"-msec");
 67 		rpc_call_request();
 68 	}
 69 
 70 	//////////////////////////////////////////////////////////////////////////
 71 	//		rpc_call						//
 72 	//////////////////////////////////////////////////////////////////////////
 73 	var rpc_call		= null;
 74 	var rpc_call_request	= function(){
 75 		// log to debug
 76 		if( verbose > 1 )	console.log("rpc: rpc_call_request enter")
 77 		// sanity check
 78 		console.assert(rpc_call === null);
 79 		// create the neoip_rpc.call
 80 		var co		= casti_opts;
 81 		rpc_call	= neoip_rpc.rpc_call.create({
 82 			call_url	: call_url,
 83 			method_name	: 'request_stream',
 84 			method_args	: [co.mdata_srv_uri, co.cast_name, co.cast_privtext, co.scasti_uri
 85 						, co.scasti_mod, co.http_peersrc_uri, co.web2srv_str],
 86 			success_cb	: function(returned_val){
 87 				rpc_call_destroy();
 88 				req_timer_refresh();
 89 				if(returned_val.length > 0){
 90 				// mark this cast as "published"
 91 					cast_privhash	= returned_val;
 92 					notify_event("ispublished", {cast_privhash: returned_val});			
 93 				}else{
 94 					// mark this cast as "not published"
 95 					cast_privhash	= null;
 96 					notify_event("nopublished", null);
 97 				}
 98 			},
 99 			failure_cb	: function(fault){
100 				// mark this cast as "not published"
101 				cast_privhash	= null;
102 				if( verbose )	console.log("failure: "+require('sys').inspect(fault));
103 				rpc_call_destroy();
104 				notify_event("rpc_error", null);
105 			}
106 		});
107 	}
108 	var rpc_call_release	= function(){
109 		// log to debug
110 		if( verbose > 1 )	console.log("rpc: rpc_call_release enter")
111 		// sanity check
112 		console.assert(rpc_call === null);
113 		// mark this cast as "not published"
114 		cast_privhash	= null;
115 		// create the neoip_rpc.call
116 		var co		= casti_opts;
117 		rpc_call	= neoip_rpc.rpc_call.create({
118 			call_url	: call_url,
119 			method_name	: 'release_stream',
120 			method_args	: [co.mdata_srv_uri, co.cast_name, co.cast_privtext],
121 			success_cb	: function(returned_val){
122 				rpc_call_destroy();
123 				notify_event("released", null);
124 			},
125 			failure_cb	: function(fault){
126 				if( verbose )	console.log("failure: "+require('sys').inspect(fault));
127 				rpc_call_destroy();
128 				notify_event("rpc_error", null);
129 			}
130 		});	
131 	}
132 	var rpc_call_destroy	= function(){
133 		if( rpc_call !== null )	rpc_call.destroy();
134 		rpc_call	= null;
135 	}
136 
137 	//////////////////////////////////////////////////////////////////////////
138 	//		notify_event						//
139 	//////////////////////////////////////////////////////////////////////////
140 	var last_event_type	= null;
141 	var last_event_data	= null;
142 	var notify_event	= function(event_type, event_data){
143 		// return if current event is equal to last event
144 		if( event_type == last_event_type && underscore.isEqual(event_data, last_event_data) )
145 			return;
146 		// backup current event_type/event_data
147 		last_event_data	= event_data;
148 		last_event_type	= event_type;
149 		// do notify the event_cb
150 		event_cb(event_type, event_data);
151 	}
152 
153 
154 	//////////////////////////////////////////////////////////////////////////
155 	//		run initialisation					//
156 	//////////////////////////////////////////////////////////////////////////
157 	// call the contructor
158 	ctor();
159 	// return the public properties
160 	return {
161 		release		: start_release,
162 		published	: function(){ return cast_privhash !== null 	},
163 		cast_privhash	: function(){ return cast_privhash;		},
164 		destroy		: dtor
165 	}
166 }
167 
168 /**
169  * Class method to create an object
170  * - thus avoid new operator
171 */
172 casti_ctrl_t.create	= function(ctor_opts){
173 	return new casti_ctrl_t(ctor_opts);
174 }
175 
176 // export it via commonjs
177 exports.create	= casti_ctrl_t.create;
