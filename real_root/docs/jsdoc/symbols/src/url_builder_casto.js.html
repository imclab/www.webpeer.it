1 /**
  2  * Create an url for neoip-casto
  3  * - opts.base_url	: casto base_url	(REQUIRED)
  4  * - opts.cast_privhash	: cast_privhash		(REQUIRED)
  5  * - opts.cast_name	: cast_name		(REQUIRED)
  6  * - opts.mdata_srv_uri	: url of the nameserver (OPTIONAL)
  7  *
  8  * - NOTE: no detection of the neoip-casto apps is done
  9  *
 10  * @return {string} the url for the stream out of neoip-casto
 11 */
 12 var create = function(opts){
 13 	// sanity check - all mandatory fields MUST be present
 14 	console.assert(opts.base_url);
 15 	console.assert(opts.cast_privhash);
 16 	console.assert(opts.cast_name);
 17 	// build the url
 18 	var url	= opts.base_url + "/" + opts.cast_privhash + "/" + opts.cast_name;
 19 	// add mdata_srv_uri if any
 20 	if( opts.mdata_srv_uri )	url	+= "?mdata_srv_uri=" + escape(opts.mdata_srv_uri);
 21 	// return the just built url
 22 	return url;
 23 }
 24 
 25 // export it via commonjs
 26 exports.create	= create;
