 15 var disp_usage	= function(prefix){
 16 	if(prefix)	console.log(prefix + "\n");
 17 	console.log("usage: neoip-url-stream [-s url] base_url cast_privhash cast_name");
 18 	console.log("");
 19 	console.log("Build an url for neoip-casto");
 20 	console.log("");
 21 	console.log("-s|--mdata_srv_uri\n\t\tSet url for the mdata_srv.");
 22 	console.log("-h|--help\tDisplay the inline help.");
 23 }
 24 var optind	= 2;
 25 for(;optind < process.argv.length; optind++){
 26 	var key	= process.argv[optind];
 27 	var val	= process.argv[optind+1];
 28 	//console.log("key="+key+" val="+val);
 29 	if( key == "-s" || key == "--mdata_srv_uri" ){
 30 		opts.mdata_srv_uri	= val;
 31 		optind		+= 1;			
 32 	}else if( key == "-h" || key == "--help" ){
 33 		disp_usage();
 34 		process.exit(0);
 35 	}else{
 36 		// if the option doesnt exist, consider it is the first non-option parameters
 37 		break;
 38 	}
 39 }
 40 // check there are at least 3 parameters remaining in the cmdline
 41 if( process.argv.length - optind < 3 ){
 42 	disp_usage("missing parameters to buidl the url");
 43 	process.exit(0);		
 44 }
 45 
 46 // get required options from the rest of the cmdline
 47 opts.base_url		= process.argv[optind++];
 48 opts.cast_privhash	= process.argv[optind++];
 49 opts.cast_name		= process.argv[optind++];
 50 
 51 // create the url
 52 var url	= url_builder_casto.create(opts);
 53 // output the url on stdout
 54 console.log(url);

 63 =head1 NAME
 64 
 65 neoip-url-stream - build a url for neoip-casto
 66 
 67 =head1 SYNOPSIS
 68 
 69 B<neoip-url-stream> [-s mdata_srv_uri] base_url cast_privhash cast_name
 70 
 71 =head1 DESCRIPTION
 72 
 73 Build an url for neoip-casto. This url will point on a normal http stream.
 74 Thus the client may be any client which understand http, it doesnt have to modified
 75 to use this url.
 76 
 77 =over
 78 
 79 =item I<base_url>
 80 
 81 The base_url is the neoip-casto base url. It is not autodetected.
 82 TODO make it autodetectable (likely with a neoip-appdetect applications
 83 or by simply make a dependancy in the code)
 84 
 85 =item I<cast_privhash>
 86 
 87 This is the hash of the broadcast password (aka casti_privtext). This is used
 88 to ensure the http client got the permission to read the broadcast.
 89 
 90 =item I<cast_name>
 91 
 92 cast_name is the name of the broadcast. This name is unique inside the I<mdata_srv_uri>.
 93 
 94 =back
 95 
 96 =head1 OPTIONS
 97 
 98 The following options are available:
 99 
100 =over
101 
102 =item B<-s|--mdata_srv_uri> I<url>
103 
104 Use to set the url for the metadata server. This server got the metadata information
105 about the broadcast. (OPTIONAL)
106 
107 =back
108 
109 =head1 EXAMPLES
110 
111 =over
112 
113 =item B<basic usage>
114 
115 $ neoip-url-stream http://localhost:4560 a761ce3a superstream
116 
117 This will display the following in stdout
118 
119 http://localhost:4560/a761ce3a/superstream
120 
121 =item B<specifying mdata_srv_uri>
122 
123 $ neoip-url-stream -s http://example.com http://localhost:4560 a761ce3a superstream
124 
125 This will display the following in stdout
126 
127 http://localhost:4560/a761ce3a/superstream?mdata_srv_uri=http%3A//example.com
